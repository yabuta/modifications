00444     void snapshotFinishedScanningBlock(TBPtr finishedBlock, TBPtr nextBlock) {
00445         if (nextBlock != NULL) {
00446             assert(m_blocksPendingSnapshot.find(nextBlock) != m_blocksPendingSnapshot.end());
00447             m_blocksPendingSnapshot.erase(nextBlock);
00448             nextBlock->swapToBucket(TBBucketPtr());
00449         }
00450         if (finishedBlock != NULL && !finishedBlock->isEmpty()) {
00451             m_blocksNotPendingSnapshot.insert(finishedBlock);
00452             int bucketIndex = finishedBlock->calculateBucketIndex();
00453             if (bucketIndex != -1) {
00454                 finishedBlock->swapToBucket(m_blocksNotPendingSnapshotLoad[bucketIndex]);
00455             }
00456         }
00457     }

00758 inline void PersistentTable::deleteTupleStorage(TableTuple &tuple, TBPtr block)
00759 {
00760     // May not delete an already deleted tuple.
00761     assert(tuple.isActive());
00762 
00763     // The tempTuple is forever!
00764     assert(&tuple != &m_tempTuple);
00765 
00766     // This frees referenced strings -- when could possibly be a better time?
00767     if (m_schema->getUninlinedObjectColumnCount() != 0) {
00768         decreaseStringMemCount(tuple.getNonInlinedMemorySize());
00769         tuple.freeObjectColumns();
00770     }
00771 
00772     tuple.setActiveFalse();
00773 
00774     // add to the free list
00775     m_tupleCount--;
00776     if (tuple.isPendingDelete()) {
00777         tuple.setPendingDeleteFalse();
00778         --m_invisibleTuplesPendingDeleteCount;
00779     }
00780 
00781     if (block.get() == NULL) {
00782         block = findBlock(tuple.address(), m_data, m_tableAllocationSize);
00783         if (block.get() == NULL) {
00784             throwFatalException("Tried to find a tuple block for a tuple but couldn't find one");
00785         }
00786     }
00787 
00788     bool transitioningToBlockWithSpace = !block->hasFreeTuples();
00789 
00790     int retval = block->freeTuple(tuple.address());
00791     if (retval != -1) {
00792         //Check if if the block is currently pending snapshot
00793         if (m_blocksNotPendingSnapshot.find(block) != m_blocksNotPendingSnapshot.end()) {
00794             //std::cout << "Swapping block " << static_cast<void*>(block.get()) << " to bucket " << retval << std::endl;
00795             block->swapToBucket(m_blocksNotPendingSnapshotLoad[retval]);
00796         //Check if the block goes into the pending snapshot set of buckets
00797         } else if (m_blocksPendingSnapshot.find(block) != m_blocksPendingSnapshot.end()) {
00798             block->swapToBucket(m_blocksPendingSnapshotLoad[retval]);
00799         } else {
00800             //In this case the block is actively being snapshotted and isn't eligible for merge operations at all
00801             //do nothing, once the block is finished by the iterator, the iterator will return it
00802         }
00803     }
00804 
00805     if (block->isEmpty()) {
00806         m_data.erase(block->address());
00807         m_blocksWithSpace.erase(block);
00808         m_blocksNotPendingSnapshot.erase(block);
00809         assert(m_blocksPendingSnapshot.find(block) == m_blocksPendingSnapshot.end());
00810         //Eliminates circular reference
00811         block->swapToBucket(TBBucketPtr());
00812     } else if (transitioningToBlockWithSpace) {
00813         m_blocksWithSpace.insert(block);
00814     }
00815 }

00817 inline TBPtr PersistentTable::findBlock(char *tuple, TBMap &blocks, int blockSize) {
00818     if (!blocks.empty()) {
00819         TBMapI i = blocks.lower_bound(tuple);
00820 
00821         // Not the first tuple of any known block, move back a block, see if it
00822         // belongs to the previous block
00823         if (i == blocks.end() || i.key() != tuple) {
00824             i--;
00825         }
00826 
00827         // If the tuple is within the block boundaries, we found the block
00828         if (i.key() <= tuple && tuple < i.key() + blockSize) {
00829             if (i.data().get() == NULL) {
00830                 throwFatalException("A block has gone missing in the tuple block map.");
00831             }
00832             return i.data();
00833         }
00834     }
00835 
00836     return TBPtr(NULL);
00837 }
