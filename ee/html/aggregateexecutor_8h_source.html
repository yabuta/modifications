00094 struct AggregateRow
00095 {
00096     void* operator new(size_t size, Pool& memoryPool, size_t nAggs)
00097     {
00098         // allocate nAggs +1 for null terminator: see resetAggs, and destructor.
00099         // Would it be cleaner to have a count data member? Not by much.
00100         return memoryPool.allocateZeroes(size + (sizeof(void*) * (nAggs + 1)));
00101     }
00102     void operator delete(void*, Pool& memoryPool, size_t nAggs) { /* NOOP -- on alloc error unroll */ }
00103     void operator delete(void*) { /* NOOP -- deallocate wholesale with pool */ }
00104 
00105     ~AggregateRow()
00106     {
00107         // Stop at the terminating null agg pointer that has been allocated as an extra and ignored since.
00108         for (int ii = 0; m_aggregates[ii] != NULL; ++ii) {
00109             // All the aggs inherit no-op delete operators, so, "delete" is really just destructor invocation.
00110             // The destructor being invoked is the implicit specialization of Agg's destructor.
00111             // The compiler generates it to invoke the destructor (if any) of the distinct value set (if any).
00112             // It must be called because the pooled Agg object only embeds the (boost) set's "head".
00113             // The "body" is allocated by boost via its defaulted stl allocator as the set grows
00114             // -- AND it is not completely deallocated when "clear" is called.
00115             // This AggregateRow destructor would not be required at all if distinct was based on a home-grown
00116             // pool-aware hash, incapable of leaking outside the pool.
00117             delete m_aggregates[ii];
00118         }
00119     }
00120 
00121     void resetAggs()
00122     {
00123         // Stop at the terminating null agg pointer that has been allocated as an extra and ignored since.
00124         for (int ii = 0; m_aggregates[ii] != NULL; ++ii) {
00125             m_aggregates[ii]->resetAgg();
00126         }
00127     }
00128 
00129     void recordPassThroughTuple(TableTuple &passThroughTupleSource, const TableTuple &tuple)
00130     {
00131         passThroughTupleSource.copy(tuple);
00132         m_passThroughTuple = passThroughTupleSource;
00133     }
00134 
00135     // A tuple from the group of tuples being aggregated. Source of pass through columns.
00136     TableTuple m_passThroughTuple;
00137 
00138     // The aggregates for each column for this group
00139     Agg* m_aggregates[0];

00327 inline AggregateExecutorBase* getInlineAggregateExecutor(const AbstractPlanNode* node) {
00328     AbstractPlanNode* aggNode = NULL;
00329     AggregateExecutorBase* aggExec = NULL;
00330     if (NULL != (aggNode = node->getInlinePlanNode(PLAN_NODE_TYPE_PARTIALAGGREGATE)) ) {
00331         VOLT_TRACE("init inline partial aggregation stuff...");
00332         aggExec = dynamic_cast<AggregatePartialExecutor*>(aggNode->getExecutor());
00333         assert(aggExec != NULL);
00334     } else if ( NULL != (aggNode = node->getInlinePlanNode(PLAN_NODE_TYPE_AGGREGATE)) ) {
00335         VOLT_TRACE("init inline serial aggregation stuff...");
00336         aggExec = dynamic_cast<AggregateSerialExecutor*>(aggNode->getExecutor());
00337         assert(aggExec != NULL);
00338     } else if (NULL != (aggNode = node->getInlinePlanNode(PLAN_NODE_TYPE_HASHAGGREGATE)) ) {
00339         VOLT_TRACE("init inline hash aggregation stuff...");
00340         aggExec = dynamic_cast<AggregateHashExecutor*>(aggNode->getExecutor());
00341         assert(aggExec != NULL);
