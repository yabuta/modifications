00152 // Output log message header in this format: [type] [file:line:function] time -
00153 // ex: [ERROR] [somefile.cpp:123:doSome()] 2008/07/06 10:00:00 -
00154 inline void outputLogHeader_(const char *file, int line, const char *func, int level) {
00155     time_t t = ::time(NULL) ;
00156     tm *curTime = localtime(&t);
00157     char time_str[32]; // FIXME
00158     ::strftime(time_str, 32, VOLT_LOG_TIME_FORMAT, curTime);
00159     const char* type;
00160     switch (level) {
00161         case VOLT_LEVEL_ERROR:
00162             type = "ERROR";
00163             break;
00164         case VOLT_LEVEL_WARN:
00165             type = "WARN ";
00166             break;
00167         case VOLT_LEVEL_INFO:
00168             type = "INFO ";
00169             break;
00170         case VOLT_LEVEL_DEBUG:
00171             type = "DEBUG";
00172             break;
00173         case VOLT_LEVEL_TRACE:
00174             type = "TRACE";
00175             break;
00176         default:
00177             type = "UNKWN";
00178     }
00179     printf("[%s] [%s:%d:%s()] %s - ", type, file, line, func, time_str);
00180 }
