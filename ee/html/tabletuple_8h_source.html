00607                                                 std::vector<char*> &oldObjects, std::vector<char*> &newObjects)

00610     assert(m_schema == source.m_schema);
00611     const int columnCount = m_schema->columnCount();
00612     const uint16_t uninlineableObjectColumnCount = m_schema->getUninlinedObjectColumnCount();
00613     /*
00614      * The source and target tuple have the same policy WRT to
00615      * inlining strings because a TableTuple used for updating a
00616      * persistent table uses the same schema as the persistent table.
00617      */

00619         uint16_t uninlineableObjectColumnIndex = 0;
00620         uint16_t nextUninlineableObjectColumnInfoIndex = m_schema->getUninlinedObjectColumnInfoIndex(0);
00621         /*
00622          * Copy each column doing an allocation for string
00623          * copies. Compare the source and target pointer to see if it
00624          * is changed in this update. If it is changed then free the
00625          * old string and copy/allocate the new one from the source.
00626          */
00627         for (uint16_t ii = 0; ii < columnCount; ii++) {
00628             if (ii == nextUninlineableObjectColumnInfoIndex) {
00629                 const TupleSchema::ColumnInfo *columnInfo = m_schema->getColumnInfo(ii);
00630                 char *       *mPtr = reinterpret_cast<char**>(getWritableDataPtr(columnInfo));
00631                 const TupleSchema::ColumnInfo *sourceColumnInfo = source.getSchema()->getColumnInfo(ii);
00632                 char * const *oPtr = reinterpret_cast<char* const*>(source.getDataPtr(sourceColumnInfo));
00633                 if (*mPtr != *oPtr) {
00634                     // Make a copy of the input string. Don't want to delete the old string
00635                     // because it's either from the temp pool or persistently referenced elsewhere.
00636                     oldObjects.push_back(*mPtr);
00637                     // TODO: Here, it's known that the column is an object type, and yet
00638                     // setNValueAllocateForObjectCopies is called to figure this all out again.
00639                     setNValueAllocateForObjectCopies(ii, source.getNValue(ii), NULL);
00640                     // Yes, uses the same old pointer as two statements ago to get a new value. Neat.
00641                     newObjects.push_back(*mPtr);
00642                 }
00643                 uninlineableObjectColumnIndex++;
00644                 if (uninlineableObjectColumnIndex < uninlineableObjectColumnCount) {
00645                     nextUninlineableObjectColumnInfoIndex =
00646                       m_schema->getUninlinedObjectColumnInfoIndex(uninlineableObjectColumnIndex);
00647                 } else {
00648                     // This is completely optional -- the value from here on has to be one that can't
00649                     // be reached by incrementing from the current value.
00650                     // Zero works, but then again so does the current value.
00651                     nextUninlineableObjectColumnInfoIndex = 0;
00652                 }
00653             } else {
00654                 // TODO: Here, it's known that the column value is some kind of scalar or inline, yet
00655                 // setNValueAllocateForObjectCopies is called to figure this all out again.
00656                 // This seriously complicated function is going to boil down to an incremental
00657                 // memcpy of a few more bytes of the tuple.
00658                 // Solution? It would likely be faster even for object-heavy tuples to work in three passes:
00659                 // 1) collect up all the "changed object pointer" offsets.
00660                 // 2) do the same wholesale tuple memcpy as in the no-objects "else" clause, below,

00668         m_data[0] = source.m_data[0];
00669     } else {
