00161         m_tail = m_head + lenPath;

00166         bool expectArrayIndex = false;

00171                 // -1 index to refer to the tail of the array

00176                         throwInvalidPathError("Unexpected termination (unterminated array access)");

00188                         break;
00189                     } else if (c < '0' || c > '9') {
00190                         throwInvalidPathError("Unexpected character in array index");
00191                     }
00192                     arrayIndex = 10 * arrayIndex + (c - '0');
00193                     if (arrayIndex > static_cast<int64_t>(INT32_MAX)) {

00195                     }

00203                         throwInvalidPathError("Invalid array index less than -1");

00218                 // a leading '.' also involves accessing the "" property of the root...
00219                 if (expectField || first) {
00220                     path.push_back(JsonPathNode(""));
00221                 }
00222                 expectField = true;
00223             } else {

00227                 do {

00234                         break;
00235                     } else if (c == '[') {
00236                         expectArrayIndex = true;
00237                         break;
