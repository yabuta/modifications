00215 inline void TempTable::deleteAllTuplesNonVirtual(bool freeAllocatedStrings) {
00216 
00217     if (m_tupleCount == 0) {
00218         return;
00219     }
00220 
00221     // Mark tuples as deleted and free strings. No indexes to update.
00222     // Don't call deleteTuple() here.
00223     const uint16_t uninlinedStringColumnCount = m_schema->getUninlinedObjectColumnCount();
00224     if (freeAllocatedStrings && uninlinedStringColumnCount > 0) {
00225         TableTuple target(m_schema);
00226         TableIterator iter(this, m_data.begin());
00227         while (iter.hasNext()) {
00228             iter.next(target);
00229             target.freeObjectColumns();
00230         }
00231     }
00232 
00233     m_tupleCount = 0;
00234     while (m_data.size() > 1) {
00235         // This block of temp table may have been clean up already
00236         // because of delete as we go feature.
00237         TBPtr blockPtr = m_data.back();
00238         m_data.pop_back();
00239         if (m_limits && blockPtr) {
00240             m_limits->reduceAllocated(m_tableAllocationSize);
00241         }
00242     }
00243 
00244     // cheap clear of the preserved first block
00245     if (!m_data.empty()) {
00246         m_data[0]->reset();
00247     }
