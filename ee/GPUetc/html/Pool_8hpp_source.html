00060 template <class T>
00061 inline T nexthigher(T k) {
00062         if (k == 0)
00063                 return 1;
00064         k--;
00065         for (int i=1; i<sizeof(T)*CHAR_BIT; i<<=1)
00066                 k = k | k >> i;
00067         return k+1;
