00108     inline int calculateBucketIndex(uint32_t tuplesPendingDeleteOnUndoRelease = 0) {
00109         if (!hasFreeTuples()) {
00110             //Completely full, don't need be considered for merging
00111             //Remove self from current bucket and null out the bucket
00112             if (m_bucket.get() != NULL) {
00113                 //std::cout << static_cast<void*>(this) << " is full, erasing from bucket" << std::endl;
00114                 m_bucket->erase(TBPtr(this));
00115                 m_bucket = TBBucketPtr();
00116             }
00117             return -1;
00118         } else if (tuplesPendingDeleteOnUndoRelease == m_activeTuples) {
00119             //Someone was kind enough to scan the whole block, move all tuples
00120             //not pending delete on undo release to another block as part
00121             //of compaction. Now this block doesn't need to be considered
00122             //for compaction anymore. The block will be completely discard along with the undo
00123             //information. Any tuples pending delete due to a snapshot will moved and picked up
00124             //by the snapshot scan from the other block
00125             if (m_bucket.get() != NULL) {
00126                 //std::cout << static_cast<void*>(this) << " has only deleted tuples that are pending undo release "
00127                 //        << tuplesPendingDeleteOnUndoRelease << std::endl;
00128                 m_bucket->erase(TBPtr(this));
00129                 m_bucket = TBBucketPtr();
00130             }
00131             return -1;
00132         }
00133         else {
00134             int index = static_cast<int>(::floor(m_activeTuples / m_tuplesPerBlockDivNumBuckets));
00135             assert(index < TUPLE_BLOCK_NUM_BUCKETS);
00136             assert(index >= 0);
00137             return index;
00138         }
00139     }

00147     inline std::pair<char*, int> nextFreeTuple() {
00148         char *retval = NULL;
00149         if (!m_freeList.empty()) {
00150             m_lastCompactionOffset = 0;
00151             retval = m_storage;
00152             TruncatedInt offset = m_freeList.back();
00153             m_freeList.pop_back();
00154             retval += offset.unpack();
00155         } else {
00156             retval = &(m_storage[m_tupleLength * m_nextFreeTuple]);
00157             m_nextFreeTuple++;
00158         }
00159         m_activeTuples++;
00160         int newBucketIndex = calculateBucketIndex();
00161         if (newBucketIndex != m_bucketIndex) {
00162             m_bucketIndex = newBucketIndex;
00163             return std::pair<char*, int>(retval, newBucketIndex);
00164         } else {
00165             return std::pair<char*, int>(retval, -1);
00166         }
