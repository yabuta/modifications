00034 static inline void micros_to_date(int64_t epoch_micros_in, boost::gregorian::date& date_out) {
00035     if (epoch_micros_in < GREGORIAN_EPOCH) {
00036         throw voltdb::SQLException(voltdb::SQLException::data_exception_numeric_value_out_of_range,
00037                 "Value out of range. Cannot convert dates prior to the year 1583");

00039     boost::posix_time::ptime input_ptime = EPOCH + boost::posix_time::microseconds(epoch_micros_in);
00040     date_out = input_ptime.date();

00066 static inline int64_t epoch_microseconds_from_components(unsigned short int year, unsigned short int month = 1,
00067         unsigned short int day = 1, int hour = 0, int minute = 0, int second = 0) {
00068     boost::gregorian::date goal_date = boost::gregorian::date(year, month, day);
00069     boost::posix_time::time_duration goal_time = boost::posix_time::time_duration(hour,minute,second);
00070     boost::posix_time::ptime goal_ptime = boost::posix_time::ptime(goal_date,goal_time);
00071     boost::posix_time::time_period goal_period (EPOCH, goal_ptime);
00072     boost::posix_time::time_duration goal_duration = goal_period.length();
00073     int64_t epoch_seconds = goal_duration.ticks() / goal_duration.ticks_per_second();
00074     return epoch_seconds * 1000000;
00075 }
