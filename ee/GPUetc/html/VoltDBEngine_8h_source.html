00567 inline int64_t VoltDBEngine::pullTuplesRemainingUntilProgressReport(AbstractExecutor* exec,
00568                                                                     Table* target_table)
00569 {
00570     if (target_table) {
00571         m_lastAccessedTable = target_table;
00572     }
00573     m_lastAccessedExec = exec;
00574     return m_tupleReportThreshold - m_tuplesProcessedSinceReport;
00575 }

00577 inline int64_t VoltDBEngine::pushTuplesProcessedForProgressMonitoring(int64_t tuplesProcessed)
00578 {
00579     m_tuplesProcessedSinceReport += tuplesProcessed;
00580     if (m_tuplesProcessedSinceReport >= m_tupleReportThreshold) {
00581         reportProgessToTopend();
00582     }
00583     return m_tupleReportThreshold; // size of next batch
00584 }
