00065     voltdb::NValue binarySearch(const int32_t hash) const {
00066         //The binary search blows up on only one range
00067         if (num_ranges == 1) {
00068             if (hash >= ranges[0].first && hash <= ranges[0].second) return NValue::getTrue();
00069             return NValue::getFalse();
00070         }
00071 
00072         /*
00073          * Bottom of a range is inclusive as well as the top. Necessary because we no longer support wrapping
00074          * from Integer.MIN_VALUE
00075          * Doing a binary search, is just a hair easier than std::lower_bound
00076          */
00077         int32_t min = 0;
00078         int32_t max = num_ranges - 1;
00079         while (min <= max) {
00080             assert(min >= 0);
00081             assert(max >= 0);
00082             uint32_t mid = (min + max) >> 1;
00083             if (ranges[mid].second < hash) {
00084                 min = mid + 1;
00085             } else if (ranges[mid].first > hash) {
00086                 max = mid - 1;
00087             } else {
00088                 return NValue::getTrue();
00089             }
00090         }
00091 
00092         return NValue::getFalse();
00093     }
