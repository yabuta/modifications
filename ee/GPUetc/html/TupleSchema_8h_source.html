00176 inline uint32_t TupleSchema::columnLengthPrivate(const int index) const {
00177     assert(index < m_columnCount);
00178     const ColumnInfo *columnInfo = getColumnInfo(index);
00179     const ColumnInfo *columnInfoPlusOne = getColumnInfo(index + 1);

00181     return static_cast<uint32_t>(columnInfoPlusOne->offset - columnInfo->offset);
