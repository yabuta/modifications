00054      * Factory method that constructs an ElasticHashinator from a binary configuration.
00055      * The format is described in ElasticHashinator.java and basically describes the tokens
00056      * on the ring.
00057      *
00058      * Config can be serialized or raw, if it is raw the pointer is stored and the data
00059      * can be shared across EEs and with Java.
00060      * The raw version consists of an array of integers where even values are tokens and odd values are partitions.
00061      *
00062      */
00063     static ElasticHashinator* newInstance(const char *config, int32_t *configPtr, uint32_t tokenCount) {
00064         if (configPtr == NULL) {
00065             ReferenceSerializeInputBE countInput(config, 4);
00066             int numEntries = countInput.readInt();
00067             ReferenceSerializeInputBE entryInput(&config[sizeof(int32_t)], numEntries * (sizeof(int32_t) + sizeof(int32_t)));
00068             int32_t *tokens = new int32_t[numEntries * 2];
00069             for (int ii = 0; ii < numEntries; ii++) {
00070                 const int32_t token = entryInput.readInt();
00071                 const int32_t partitionId = entryInput.readInt();
00072                 tokens[ii * 2] = token;
00073                 tokens[ii * 2 + 1] = partitionId;
00074             }
00075             return new ElasticHashinator(tokens, numEntries, true);
00076         } else {
00077             return new ElasticHashinator(configPtr, tokenCount, false);
00078         }
00079     }

00116     int32_t partitionForToken(int32_t hash) const {
00117         int32_t min = 0;
00118         int32_t max = tokenCount - 1;
00119 
00120         while (min <= max) {
00121             assert(min >= 0);
00122             assert(max >= 0);
00123             uint32_t mid = (min + max) >> 1;
00124             int32_t midval = tokens[mid * 2];

00126             if (midval < hash) {
