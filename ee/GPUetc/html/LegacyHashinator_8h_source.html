00059    int32_t hashinate(int64_t value) const {

00073    int32_t hashinate(const char *string, int32_t length) const {
00074        int32_t hashCode = 0;
00075        int32_t offset = 0;
00076        if (length < 0) {

00079        for (int32_t ii = 0; ii < length; ii++) {
00080           hashCode = 31 * hashCode + string[offset++];

00082        return abs(hashCode % partitionCount);
