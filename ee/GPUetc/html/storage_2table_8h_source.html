00098     /*
00099      * Table lifespan can be managed bya reference count. The
00100      * reference is trivial to maintain since it is only accessed by
00101      * the execution engine thread. Snapshot, Export and the
00102      * corresponding CatalogDelegate may be reference count
00103      * holders. The table is deleted when the refcount falls to
00104      * zero. This allows longer running processes to complete
00105      * gracefully after a table has been removed from the catalog.
00106      */
00107     void incrementRefcount() {
00108         m_refcount += 1;
00109     }
00110 
00111     void decrementRefcount() {
00112         m_refcount -= 1;
00113         if (m_refcount == 0) {
00114             delete this;
00115         }
00116     }

00380     bool compactionPredicate() {
00381         //Unfortunate work around for the fact that multiple undo quantums cause this to happen

00388     }
