00060 /*
00061  * Convert from a uint64_t that has had a signed number packed into it to
00062  * the specified signed type. The max signed value for that type is supplied as
00063  * the typeMaxValue template parameter.
00064  */
00065 template<typename signedType, int64_t typeMaxValue>
00066 inline static signedType convertUnsignedValueToSignedValue(uint64_t value) {
00067     int64_t retval = static_cast<int64_t>(value);
00068     retval -= typeMaxValue + 1;
00069     return static_cast<signedType>(retval);
00070 }

00072 /*
00073  * Specialization of convertUnsignedValueToSignedValue for int64_t. int64_t requires a comparison
00074  * to prevent overflow.
00075  */
00076 template<>
00077 inline int64_t convertUnsignedValueToSignedValue< int64_t, INT64_MAX>(uint64_t value) {
00078     if (value > static_cast<uint64_t>(INT64_MAX) + 1) {
00079         value -= INT64_MAX;
00080         value--;
00081         return static_cast<int64_t>(value);
00082     } else {
00083         int64_t retval = static_cast<int64_t>(value);
00084         retval -= INT64_MAX;
00085         retval--;
00086         return retval;
00087     }
00088 }

00136     /*
00137      * Take a value that is part of the key (already converted to a uint64_t) and inserts it into the
00138      * most significant bytes available in the key. Templated on the size of the type of key being inserted.
00139      * This allows the compiler to unroll the loop.
00140      *
00141      *
00142      * Algorithm is:
00143      * Start with the most significant byte of the keyValue we are inserting and put into the most significant byte
00144      * of the uint64_t portion of the key that is indexed by keyOffset. The most significant available byte within
00145      * the uint64_t portion of the key is indexed by intraKeyOffset. Both keyOffset and intraKeyOffset are passed
00146      * by reference so they can be updated.
00147      *
00148      *
00149      * Rinse and repeat with the less significant bytes.
00150      *
00151      */
00152     template <typename keyValueType>
00153     inline void insertKeyValue(int &keyOffset, int &intraKeyOffset, uint64_t keyValue) {
00154         for (int ii = static_cast<int>(sizeof(keyValueType)) - 1; ii >= 0; ii--) {
00155 
00156             /*
00157              * Extract the most significant byte from keyValue by shifting it all the way to the right.
00158              * Mask off the rest. Then shift it to the left to the most significant byte location available
00159              * in the key and OR it in.
00160              */
00161             data[keyOffset] |= (0xFF & (keyValue >> (ii * 8))) << (intraKeyOffset * 8); //
00162             intraKeyOffset--;//Move the offset inside the uint64 key back one.
00163             /*
00164              * If there are no more bytes available in the uint64_t indexed by keyOffset then increment keyOffset
00165              * to point to the next uint64_t and set intraKeyOffset to index to the most significant byte
00166              * in this next uint64_t.
00167              */
00168             if (intraKeyOffset < 0) {
00169                 intraKeyOffset = static_cast<int>(sizeof(uint64_t) - 1);
00170                 keyOffset++;
00171             }
00172         }
00173     }

00175     /*

00186                 keyOffset++;

00532     {

00534         // so that data memory management becomes a simple matter of running the expected

00536         // its previous value (presumably obsolete) will go out of scope with other!

00542         // Exactly one full-blown key must own each tuple and its objects.
00543         // So either use other as a lifeboat for the prior value of *this.
00544         // It's destructor will deal with the old value of "this", probably VERY SOON.
00545         // Or mark it as ephemeral if that's what *this was.
00546         GenericPersistentKey& writableOther = const_cast<GenericPersistentKey&>(other);
00547 
00548         if (keptKeySchema) {
00549             writableOther.m_keySchema = keptKeySchema;
00550             // Other needs to lifeboat the original value of *this.
00551             char keptData[keySize];
00552             ::memcpy(keptData, this->data, keySize);
00553 
00554             ::memcpy(this->data, other.data, keySize);
00555

00647 /*
00648  * TupleKey is the all-purpose fallback key for indexes that can't be
00649  * better specialized. Each TupleKey wraps a pointer to a *persistent
00650  * table tuple*. TableIndex knows the column indices from the
00651  * persistent table that form the index key. TupleKey uses this data
00652  * to evaluate and compare keys by extracting and comparing
00653  * the appropriate columns' values.
00654  *
00655  * Note that the index code will create keys in the schema of the
00656  * the index key. While all TupleKeys resident in the index itself
00657  * will point to persistent tuples, there are ephemeral TupleKey
00658  * instances that point to tuples in the index key schema.
00659  *
00660  * Pros: supports any combination of columns in a key. Each index
00661  * key is 24 bytes (a pointer to a tuple and a pointer to the column
00662  * indices (which map index columns to table columns).

00664  * Cons: requires an indirection to evaluate a key (must follow the
